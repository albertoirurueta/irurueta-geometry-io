299     @Test
300     void testLoadAndIterate() throws IllegalArgumentException, LockedException, NotReadyException, IOException,
301             LoaderException, NotAvailableException {
302 
303         final var maxNumberOfVerticesInChunk = 500;
304         final var fileStlBin = new File("./src/test/java/com/irurueta/geometry/io/pitcher-bin.stl");
305         final var fileStlAscii = new File("./src/test/java/com/irurueta/geometry/io/pitcher-ascii.stl");
306 
307         try (final var binLoader = new LoaderSTL(fileStlBin, maxNumberOfVerticesInChunk);
308              final var asciiLoader = new LoaderSTL(fileStlAscii, maxNumberOfVerticesInChunk)) {
309             binLoader.setListener(this);
310             asciiLoader.setListener(this);
311 
312             final var binIt = binLoader.load();
313 
314             assertNull(binLoader.getSolidName());
315             assertNotNull(binLoader.getNumberOfVertices());
316 
317             assertTrue(isEndValid());
318             assertTrue(isLockedValid());
319             assertTrue(isProgressValid());
320             assertTrue(isStartValid());
321             resetListener();
322 
323             final var asciiIt = asciiLoader.load();
324 
325             assertNotNull(asciiLoader.getSolidName());
326             assertNotNull(asciiLoader.getNumberOfVertices());
327             assertEquals(0L, asciiLoader.getNumberOfVertices().longValue());
328 
329             assertTrue(isEndValid());
330             assertTrue(isLockedValid());
331             assertTrue(isProgressValid());
332             assertTrue(isStartValid());
333             resetListener();
334 
335             // check correctness of chunks
336             while (binIt.hasNext() && asciiIt.hasNext()) {
337                 final var binChunk = binIt.next();
338 
339                 assertTrue(isEndValid());
340                 assertTrue(isLockedValid());
341                 assertTrue(isProgressValid());
342                 assertTrue(isStartValid());
343                 resetListener();
344 
345                 final var asciiChunk = asciiIt.next();
346 
347                 assertTrue(isEndValid());
348                 assertTrue(isLockedValid());
349                 assertTrue(isProgressValid());
350                 assertTrue(isStartValid());
351                 resetListener();
352 
353                 final var binVertices = binChunk.getVerticesCoordinatesData();
354                 final var binTexture = binChunk.getTextureCoordinatesData();
355                 final var binColors = binChunk.getColorData();
356                 final var binNormals = binChunk.getNormalsData();
357                 final var binIndices = binChunk.getIndicesData();
358 
359                 final var asciiVertices = asciiChunk.getVerticesCoordinatesData();
360                 final var asciiTexture = asciiChunk.getTextureCoordinatesData();
361                 final var asciiColors = asciiChunk.getColorData();
362                 final var asciiNormals = asciiChunk.getNormalsData();
363                 final var asciiIndices = asciiChunk.getIndicesData();
364 
365                 assertEquals(binVertices.length, binNormals.length);
366                 assertNull(binTexture);
367                 assertNull(binColors);
368                 assertEquals(asciiVertices.length, asciiNormals.length);
369                 assertNull(asciiTexture);
370                 assertNull(asciiColors);
371 
372                 final var nVerticesInChunk = binVertices.length / 3;
373 
374                 assertEquals(nVerticesInChunk, binVertices.length / 3);
375                 assertEquals(nVerticesInChunk, asciiVertices.length / 3);
376 
377                 for (var i = 0; i < nVerticesInChunk; i++) {
378                     // check x coordinate
379                     assertEquals(binVertices[3 * i], asciiVertices[3 * i], ERROR);
380 
381                     // check y coordinate
382                     assertEquals(binVertices[3 * i + 1], asciiVertices[3 * i + 1], ERROR);
383 
384                     // check z coordinate
385                     assertEquals(binVertices[3 * i + 2], asciiVertices[3 * i + 2], ERROR);
386 
387                     // check x normal
388                     assertEquals(binNormals[3 * i], asciiNormals[3 * i], ERROR);
389 
390                     // check y normal
391                     assertEquals(binNormals[3 * i + 1], asciiNormals[3 * i + 1], ERROR);
392 
393                     // check z normal
394                     assertEquals(binNormals[3 * i + 2], asciiNormals[3 * i + 2], ERROR);
395                 }
396 
397                 final var nIndicesInChunk = binIndices.length;
398                 assertEquals(nIndicesInChunk, asciiIndices.length);
399 
400                 for (var i = 0; i < nIndicesInChunk; i++) {
401                     assertTrue(binIndices[i] < nVerticesInChunk);
402                     assertTrue(asciiIndices[i] < nVerticesInChunk);
403 
404                     assertEquals(binIndices[i], asciiIndices[i]);
405                 }
406 
407                 // check bounding box values
408                 assertEquals(binChunk.getMinX(), asciiChunk.getMinX(), ERROR);
409                 assertEquals(binChunk.getMinY(), asciiChunk.getMinY(), ERROR);
410                 assertEquals(binChunk.getMinZ(), asciiChunk.getMinZ(), ERROR);
411                 assertEquals(binChunk.getMaxX(), asciiChunk.getMaxX(), ERROR);
412                 assertEquals(binChunk.getMaxY(), asciiChunk.getMaxY(), ERROR);
413                 assertEquals(binChunk.getMaxZ(), asciiChunk.getMaxZ(), ERROR);
414 
415                 assertTrue(binChunk.getMinX() <= binChunk.getMaxX());
416                 assertTrue(binChunk.getMinY() <= binChunk.getMaxY());
417                 assertTrue(binChunk.getMinZ() <= binChunk.getMaxZ());
418                 assertTrue(asciiChunk.getMinX() <= asciiChunk.getMaxX());
419                 assertTrue(asciiChunk.getMinY() <= asciiChunk.getMaxY());
420                 assertTrue(asciiChunk.getMinZ() <= asciiChunk.getMaxZ());
421             }
422 
423             if (binIt.hasNext()) {
424                 fail("Wrong number of chunks in STL");
425             }
426             if (asciiIt.hasNext()) {
427                 fail("Wrong number of chunks in PLY");
428             }
429         }
430     }

432     @Override
433     public void onLoadStart(final Loader loader) {
434         if (startCounter != 0) {
435             startValid = false;
436         }
437         startCounter++;
438 
439         checkLocked((LoaderSTL) loader);
440     }
441 
442     @Override
443     public void onLoadEnd(final Loader loader) {
444         if (endCounter != 0) {
445             endValid = false;
446         }
447         endCounter++;
448 
449         checkLocked((LoaderSTL) loader);
450     }
451 
452     @Override
453     public void onLoadProgressChange(final Loader loader, final float progress) {
454         if ((progress < 0.0) || (progress > 1.0)) {
455             progressValid = false;
456         }
457         if (progress < previousProgress) {
458             progressValid = false;
459         }
460         previousProgress = progress;
461 
462         checkLocked((LoaderSTL) loader);
463     }
464 
465     private void checkLocked(final LoaderSTL loader) {
466         if (!loader.isLocked()) {
467             lockedValid = false;
468         }
469 
470         try {
471             loader.setListener(null);
472             lockedValid = false;
473         } catch (final LockedException ignore) {
474             // no action needed
475         } catch (final Throwable e) {
476             lockedValid = false;
477         }
478 
479         try {
480             loader.load();
481             lockedValid = false;
482         } catch (final LockedException ignore) {
483             // no action needed
484         } catch (final Throwable e) {
485             lockedValid = false;
486         }
487 
488         try {
489             loader.setMaxVerticesInChunk(1);
490             lockedValid = false;
491         } catch (final LockedException ignore) {
492             // no action needed
493         } catch (final Throwable e) {
494             lockedValid = false;
495         }
496     }
497 
498     private void resetListener() {
499         startValid = endValid = progressValid = lockedValid = true;
500         startCounter = endCounter = 0;
501         previousProgress = 0.0f;
502     }
503 
504     private boolean isStartValid() {
505         return startValid;
506     }
507 
508     private boolean isEndValid() {
509         return endValid;
510     }
511 
512     private boolean isProgressValid() {
513         return progressValid;
514     }
515 
516     private boolean isLockedValid() {
517         return lockedValid;
