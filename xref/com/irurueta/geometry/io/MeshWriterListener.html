16  package com.irurueta.geometry.io;
17  
18  import java.io.File;
19  
20  /**
21   * Listener for a mesh writer. This listener will notify when writing starts,
22   * finishes, notifies progress changes, when textures need to be validated, when
23   * a chunk of data has been loaded, etc.
24   */
25  public interface MeshWriterListener {
26  
27      /**
28       * Notifies when writing starts.
29       *
30       * @param writer reference to writer.
31       */
32      void onWriteStart(final MeshWriter writer);
33  
34      /**
35       * Notifies when writing ends.
36       *
37       * @param writer reference to writer.
38       */
39      void onWriteEnd(final MeshWriter writer);
40  
41      /**
42       * Notifies progress changes during writing.
43       *
44       * @param writer   reference to writer.
45       * @param progress progress amount as a value between 0.0 and 1.0.
46       */
47      void onWriteProgressChange(final MeshWriter writer, final float progress);
48  
49      /**
50       * Called when a chunk of data is available.
51       *
52       * @param writer reference to writer.
53       * @param chunk  chunk of data that has been loaded.
54       */
55      void onChunkAvailable(final MeshWriter writer, final DataChunk chunk);
56  
57      /**
58       * Called when a material file is needed. A path used as a hint to locate
59       * such file will be provided, and using that path the File must be
60       * retrieved (by any means, such as downloading from a remote location or
61       * uncompressing from a file, etc.).
62       * Returned file will be handled by the callee of this method. If a
63       * temporary file is created copying the contents of the file from a remote
64       * location, then it will be the responsibility of the callee to delete the
65       * temporary file.
66       *
67       * @param writer reference to writer.
68       * @param path   path where material file is supposed to be found.
69       * @return File containing the material file.
70       */
71      File onMaterialFileRequested(final MeshWriter writer, final String path);
72  
73      /**
74       * Called when a texture needs to be validated.
75       * Provided texture data will contain the path that can be used as a hint to
76       * locate the texture image. The texture image file can be retrieved by
77       * any means, such as downloading form a remote location or uncompressing
78       * from a file, etc.
79       * Returned file will be handled by the callee of this method. If a
80       * temporary file is created copying the contents of the original texture
81       * (or modifying such texture by transcoding it or resizing it), then it
82       * will be the responsibility of the callee to delete the temporary file.
83       *
84       * @param writer  reference to writer.
85       * @param texture reference to texture.
86       * @return File containing texture to be validated.
87       */
88      File onValidateTexture(final MeshWriter writer, final Texture texture);
89  
90      /**
91       * Called when texture has been validated.
92       * This method is called so that the callee has a chance to remove the
93       * texture file if needed.
94       *
95       * @param writer reference to writer.
96       * @param f      reference to texture file that has been validated. If this file
97       *               was meant to be temporary, this method gives a chance to remove it
98       *               because it will no longer be needed.
99       */
100     void onDidValidateTexture(final MeshWriter writer, final File f);
101 
102     /**
103      * Called when texture data is found in a binary file.
104      * This method should create a temporary file where such data will be
105      * copied to.
106      *
107      * @param writer        reference to writer.
108      * @param textureWidth  texture image width in pixels.
109      * @param textureHeight texture image height in pixels.
110      * @return File where texture data will be copied to.
111      */
112     File onTextureReceived(final MeshWriter writer, final int textureWidth, final int textureHeight);
113 
114     /**
115      * Called when texture data of a binary file has been copied to temporal
116      * file and is available for processing.
117      * This method gives the callee a chance to transcode, transform or resize
118      * the texture image as needed.
119      *
120      * @param writer        reference to writer.
121      * @param textureFile   file that contains texture image data read from stream.
122      * @param textureWidth  texture image width in pixels.
123      * @param textureHeight texture image height in pixels.
124      * @return converted image trans-coded into JPG format or resized if
125      * needed, otherwise textureFile can be returned if no changes are needed.
126      */
127     File onTextureDataAvailable(final MeshWriter writer, final File textureFile, final int textureWidth,
128                                 final int textureHeight);
129 
130     /**
131      * Called when texture data has already been processed for a binary file.
132      * This method gives a chance to remove any temporal texture image as it
133      * will no longer be needed.
134      *
135      * @param writer        reference to writer.
136      * @param textureFile   file that contains texture image data that can be
137      *                      removed.
138      * @param textureWidth  texture image width in pixels.
139      * @param textureHeight texture image height in pixels.
140      */
141     void onTextureDataProcessed(final MeshWriter writer, final File textureFile, final int textureWidth,
142                                 final int textureHeight);
143 }
