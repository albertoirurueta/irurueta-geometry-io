16  package com.irurueta.geometry.io;
17  
18  import java.io.File;
19  
20  public interface LoaderListenerBinary extends LoaderListener {
21  
22      /**
23       * Callee should create a file where texture data will be stored.
24       * Caller of this method will be in charge of writing texture data in JPG
25       * format at the file returned by this method.
26       *
27       * @param loader             Caller of this method.
28       * @param textureId          internal identifier of texture.
29       * @param textureImageWidth  Texture image width in pixels.
30       * @param textureImageHeight Texture image height in pixels.
31       * @return A file where texture data will be stored.
32       */
33      File onTextureReceived(final LoaderBinary loader, final int textureId, final int textureImageWidth,
34                             final int textureImageHeight);
35  
36      /**
37       * Indicates that texture data has been stored in provided texture file,
38       * which is the one returned by onTextureReceived(LoaderBinary).
39       * Texture data is always provided in JPG format.
40       *
41       * @param loader             Caller of this method.
42       * @param textureFile        File where texture data has been stored.
43       * @param textureId          id assigned to texture.
44       * @param textureImageWidth  Texture image width in pixels.
45       * @param textureImageHeight Texture image height in pixels.
46       * @return True to indicate that texture data is a valid image,
47       * false otherwise. Returning false will halt the loading process for a
48       * binary loader by raising a LoaderException.
49       * @see #onTextureReceived(LoaderBinary, int, int, int)
50       */
51      boolean onTextureDataAvailable(final LoaderBinary loader, final File textureFile, final int textureId,
52                                     final int textureImageWidth, final int textureImageHeight);
53  }
